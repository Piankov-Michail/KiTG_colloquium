// Поиск индуцированного полного двудольного подграфа K_{2,2}
for u in V:
    // Пропуск, если есть петля у u
    if u ∈ adj[u]:
        CONTINUE
    for v in V, v > u: // Сравнивается номер вершины, чтобы не повторяться
        // Пропуск, если u и v смежны или есть петля у v   
        if v ∈ adj[u] или v ∈ adj[v]:  
            CONTINUE
        
        // Нахождение общих соседей u и v
        common = adj[u] ∩ adj[v]
        if |common| < 2:
            CONTINUE
        
        // Перебор пар общих соседей
        for w in common:
            // Пропуск, если есть петля у w
            if w ∈ adj[w]:
                CONTINUE
            for z in common, z > w:
                // Если w и z не смежны и у z нет петли, то такой вариант подходит
                if z ∉ adj[w] и не(z ∈ adj[z]):
